public partial class MainWindow : Window
{
    public ObservableCollection<Employee> employees { get; set; }
    private int sortBy = 0;

    public MainWindow()
    {
        InitializeComponent();
        employees = new ObservableCollection<Employee>();
        dataGrid.ItemsSource = employees;
    }
    private void SortButton_Click(object sender, RoutedEventArgs e)
    {
        sortBy = (sortBy + 1) % 3;
        if (sortBy == 0)
        {
            employees = new ObservableCollection<Employee>(employees.OrderBy(x => x.Name));
        }
        else if(sortBy == 1)
        {
            employees = new ObservableCollection<Employee>(employees.OrderBy(x => x.ID));
        }
        else
        {
            employees = new ObservableCollection<Employee>(employees.OrderBy(x => x.Speciality));
        }
        dataGrid.ItemsSource = employees;
    }
}
